# Filter out cells that: 
## Express > 10,000 genes (could indicate doublets)
## Express > 100,000 transcripts (could indicate doublets)
## Have > 0.25 doublet score by scrublet
## Express > 20% mitochondrial RNAs (could indicate dead cells)
def filter_genes_counts_mt(anndata): 
    "anndata: AnnData object to filter"
    X = anndata.obs
    # Hard cut-off indicated here:
    nGene_cutOff  = 10000
    nCount_cutOff = 100000
    mito_cutOff   = np.quantile(X['pct_counts_mt'], 0.995).round(0)
    scrub_cutOff  = 0.25
    # 20% should be the lower threshold for mtDNA 
    if mito_cutOff > 20:
        mito_cutOff = 20
    mask = (X['n_genes_by_counts'] < nGene_cutOff) & \
           (X['n_genes_by_counts'] > 500) & \
           (X['total_counts'] < nCount_cutOff) & \
           (X['pct_counts_mt'] < mito_cutOff) & \
           (X['doublet_score'] < scrub_cutOff)
    print('Filtering cells with nGenes {} < genes < {}'.format(500, nGene_cutOff))
    print('Filtering cells with nCounts <', nCount_cutOff)
    print('Filtering cells with mitoPct <', mito_cutOff)
    print('Filtering cells with doubletscore <', scrub_cutOff)
    bdata = anndata[mask].copy()
    return bdata
